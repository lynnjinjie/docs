# const 常量

// const 定义的是常量,只读,但是它指向的地址可以修改
const s = [1, 2, 3]
s[0] = 0
// s = [0,2,3]

# 解构赋值

var obj = { x: 1, y: 2, z: 3 }
const { x: a, y: b, z: c } = obj
//a=1,b=2,c=3

# 嵌套解构

const obj = { start: { x: 1, y: 2 } }
const {
  start: { x: a, y: b },
} = obj
// a=1,b=2

# 数组解构

const [a, , , b] = [1, 2, 3, 4, 5]
// a = 1,b=4
//rest操作符只能放在数组末尾
const [a, b, ...arr] = [1, 2, 3, 4, 5]
// arr = [3,4,5]

# class 类

class book {
  constructor(name) {
    this.name = name
  }
  // Getter 函数的作用是可以让返回一个对象私有变量的值给用户，而不需要直接去访问私有变量。
  get writer() {
    return this.name
  }
  // Setter 函数的作用是可以基于传进的参数来修改对象中私有变量的值。这些修改可以是计算，或者是直接替换之前的值。
  set writer(updatedName) {
    this.name = updatedName
  }
}

const lol = new book('english')
// lol.writer = english
lol.writer = 'wut'
// lol.writer = wut
